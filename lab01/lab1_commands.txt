# Navigate to home and create a directory
cd /home
mkdir my_dir
cd my_dir

# Create a file and give permissions
touch my_file.txt
chmod 777 my_file.txt
cat my_file.txt

# Go back to parent and create another directory
cd ..
mkdir another_dir

# Copy the file to another directory and remove original
cp my_dir/my_file.txt another_dir/
rm my_dir/my_file.txt

# Verify file in new directory
cd another_dir
ls

# Create two directories and move files
cd ~
mkdir dir1 dir2
cd dir1
touch file1.txt file2.txt file3.txt
cd ..
mv dir1/* dir2/
ls dir2
